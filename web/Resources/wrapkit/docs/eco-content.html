Wrapkit Content

Usage

var myContent = new WrapkitContent( '#myContent', options );

Options

Name Type Default Descriptions
rtlMode boolean false set the Content direction to rtl/ltr.

setter

var opts = {
  rtlMode: false
};
myContent.option( opts );

getter

var options = myContent.options;
get individually
var isRtl = myContent.options.rtlMode;

Methods

rtl()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set rtl direction
  myLayout.rtl(true);
});

Events

init

myContent.on( "init", function( myContent ) { });

rtl

myContent.on( "rtl", function( myContent ) { });
