Wrapkit Header

Usage

var myHeader = new WrapkitHeader( '#myHeader', options );

Options

Name Type Default Descriptions
fixed boolean false set the Header to fixed position.
fixedPosition string top set the Header fixed position to top/bottom.
rtlMode boolean false set the Header direction to rtl/ltr.
skin string default

set the Header skin. Available skins:

default, inverse, light, red, orange, yellow, green, teal, cyan, blue, violet.

setter

var opts = {
  fixed: false,
  fixedPosition: "top",
  rtlMode: false,
  skin: "default"
};
myHeader.option( opts );

getter

var options = myHeader.options;
get individually
var skin = myHeader.options.skin;

Methods

setSkin()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set header skin
  myHeader.setSkin('inverse');
});

fixed()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // toggle header to fixed
  myHeader.fixed(true);
});

fixedTop()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set header fixed to top
  myHeader.fixedTop();
});

fixedBottom()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set header fixed to bottom
  myHeader.fixedBottom();
});

rtl()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set header direction
  myHeader.rtl(true);
});
