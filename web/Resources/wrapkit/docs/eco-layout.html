Wrapkit Layout

wrapkitLayout work on .wrapkit-wrapper lets you set the basic layout of the application.

Usage

var myLayout = new WrapkitLayout( '#myLayout', options );

Options

Name Type Default Descriptions
box boolean false set the Layout to fluid/box.
fullscreen boolean false set the Layout to use fullscreen or not.

setter

var opts = {
  box: true,
  fullscreen: false
};
myLayout.option( opts );

getter

var options = myLayout.options;
get individually
var isLayoutBox = myLayout.options.box;

Methods

setFluid()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set to fluid
  myLayout.setFluid();
});

setBox()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set to box
  myLayout.setBox();
});

fullscreen()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // make fullscreen
  myLayout.fullscreen();
});

exitFullscreen()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // exit fullscreen
  myLayout.exitFullscreen();
});

myLayout.on( "init", function( myLayout ) { });
