Wrapkit Panel

Usage

var myPanel = new WrapkitPanel( '#myPanel', options );

Options

Name Type Default Descriptions
collapse boolean false toggle collapse a Panel.
expand boolean false toggle expand a Panel. It's make a Panel to fit document sizes.
loaderTemplate string <i class='fa fa-spin fa-spinner'></i> the loader indicator (icon) that use when panel loader is show.
loaderColor string blue

color that applied to the loaderTemplate. Available colors:

dark, light, red, orange, yellow, green, teal, cyan, blue, violet.
context string default contextual color of panel.
fillColor boolean false activated Panel fill color.

setter

var opts = {
  collapse: false,
  expand: false,
  loaderTemplate: "<i class='fa fa-spin fa-spinner'></i>",
  context: 'default',
  loaderColor: 'blue',
  fillColor: false
};
myPanel.option( opts );

getter

var options = myPanel.options;
get individually
var isExpand = myPanel.options.expand;

Methods

setContext()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set contextual color
  myPanel.setContext('teal');
});

setFill()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // use fill color
  myPanel.setFill(true);
});

hide()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // hidden panel
  myPanel.hide();
});

collapse()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // toggle collapse a panel
  myPanel.collapse(true);
});

expand()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // toggle expand a panel
  myPanel.expand(true);
});

showLoader()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // show loader indicator
  myPanel.showLoader();
});
