Wrapkit Sidebar

Usage

var mySidebar = new WrapkitSidebar( '#mySidebar', options );

Options

Name Type Default Descriptions
align string left set the Sidebar position to left/right (not including direction).
caret object 
{
  collapse: "fa-angle-right",
  expand: "fa-angle-down",
  prefix: "fa"
}
 the icon open/collapse indicator on sidebar menu. You can use all of available icons.
context string teal Contextual color of Sidebar menu.
fixed boolean false set the Sidebar to fixed.
loader string fa-spinner loader indicator, useful for server side actions (save, edit, delete, load, etc) like ajax or something else.
maxResize int 360 will give a value for the maximal Sidebar size that can be done with resizable (unit in px).
minResize int 220 will give a value for the minimum Sidebar size that can be done with resizable (unit in px).
mode string vertical rendering sidebar as a vertical or horizontal line.
resizable boolean true allow the users to resize the sidebar with mouse/touch event.
rtlMode boolean false set the Sidebar direction to rtl/ltr.
size string lg variant of Sidebar size (lg/sm).
rtlMode boolean false set the Sidebar direction to rtl/ltr.
skin string dark

set the Sidebar skin. Available skins:

default, dark, light, red, orange, yellow, green, teal, cyan, blue, violet.
variant string tabs variant of sidebar menu style (pills/tabs).
visible boolean true hide/show the Sidebar.

setter

var opts = {
  align: "left",
  caret: {
    collapse: "fa-angle-right",
    expand: "fa-angle-down",
    prefix: "fa"
  },
  context: "teal",
  fixed: false,
  loader: "fa-spinner",
  maxResize: 360,
  minResize: 220,
  mode: "vertical",
  resizable: true,
  rtlMode: false,
  size: "lg",
  skin: "dark",
  variant: "tabs",
  visible: true
};
mySidebar.option( opts );

getter

var options = mySidebar.options;
get individually
var isShow = mySidebar.options.visibility;

Methods

setMode()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // use horizontal mode
  mySidebar.setMode("horizontal");
});

setSkin()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set the sidebar skin
  mySidebar.setSkin("light");
});

setContext()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set the sidebar context
  mySidebar.setContext("red");
});

setVariant()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set the sidebar variant
  mySidebar.setVariant("pills");
});

fixed()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set the sidebar to fixed position
  mySidebar.fixed(true);
});

size()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // resize the sidebar to small size
  mySidebar.size("sm");
});

hide()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // hidden the sidebar
  mySidebar.hide();
});

show()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // show the sidebar
  mySidebar.show();
});

showLoader()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // show event loader
  var index = mySidebar.elem.querySelector('.nav-item:eq(3)');
  mySidebar.showLoader( index, 'fa-spinner' );
});

hideLoader()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // show event loader
  var index = mySidebar.elem.querySelector('.nav-item:eq(3)');
  mySidebar.hideLoader( index );
});

align()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set the sidebar align
  mySidebar.align("right");
});

rtl()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // set the sidebar direction
  mySidebar.rtl(true);
});

resizable()

document.querySelector("#myButton")
  .addEventListener( "click", function(){
  // turn on/off mouse resizable
  mySidebar.resizable(false);
});
